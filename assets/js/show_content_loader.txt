function displayShowName(shows) {
    var titleElt = document.getElementsByTagName("h1")[0];
    titleElt.textContent = shows.results[0].name;

    var backgroundElt = document.getElementsByClassName("show_layer")[0]
    var baseUrl = 'https://image.tmdb.org/t/p'
    backgroundElt.style.backgroundImage = "url('" + baseUrl + shows.results[0].backdrop_path + "')"
}

window.addEventListener('load', () => {
    fetch("https://api.themoviedb.org/3/search/tv?api_key=9681493c16e2c16cba85aee9de76d451&language=fr-FR&page=1&include_adult=false&query=lucifer")
    .then(response => response.json())
    .then(data => displayShowName(data))
    .catch(error => console.log(error));
})
